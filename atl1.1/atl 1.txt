import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from pandas.plotting import andrews_curves, parallel_coordinates, radviz

sns.set(style="whitegrid", color_codes=True)

df = pd.read_csv("flower_dataset.csv")

print("First 5 rows of dataset:")
print(df.head())
print("\nSpecies distribution:")
print(df["species"].value_counts())

# 1. Scatter Plot
sns.scatterplot(data=df, x="sepal_length", y="sepal_width", hue="species")
plt.title("Scatter Plot: Sepal Length vs Width by Species")

# 2. Joint Plot
sns.jointplot(data=df, x="sepal_length", y="sepal_width", hue="species")
plt.suptitle("Joint Plot of Sepal Length and Width", y=1.02)

# 3. FacetGrid Scatter Plot
sns.FacetGrid(df, hue="species", height=5) \
   .map_dataframe(sns.scatterplot, x="sepal_length", y="sepal_width") \
   .add_legend()
plt.title("FacetGrid Scatter Plot by Species")

# 4. Box Plot
sns.boxplot(x="species", y="petal_length", data=df)
plt.title("Box Plot: Petal Length by Species")

# 5. Violin Plot
sns.violinplot(x="species", y="petal_length", data=df)
plt.title("Violin Plot: Petal Length by Species")

# 6. KDE Plot
sns.kdeplot(data=df, x="petal_length", hue="species", fill=True)
plt.title("KDE Plot: Petal Length by Species")

# 7. Pair Plot
sns.pairplot(df, hue="species", corner=True, diag_kind="kde")
plt.suptitle("Pair Plot of Flower Dataset", y=1.02)

# 8. Box Plot Grouped by Species
df.boxplot(by="species", figsize=(10, 6))
plt.suptitle("Box Plot Grouped by Species")

# 9. Andrews Curves
plt.figure(figsize=(10, 5))
andrews_curves(df, "species", color=sns.color_palette("Set2"))
plt.title("Andrews Curves")

# 10. Parallel Coordinates
plt.figure(figsize=(10, 5))
parallel_coordinates(df, "species", color=sns.color_palette("Dark2"))
plt.title("Parallel Coordinates Plot")

# 11. Radviz Plot
plt.figure(figsize=(8, 8))
radviz(df, "species", color=sns.color_palette("tab10"))
plt.title("Radviz Plot")
